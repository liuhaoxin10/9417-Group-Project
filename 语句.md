# Q1:--------------------------------------------
```sql
SELECT p.unswid, p.name
FROM people p
JOIN students s ON s.id = p.id
JOIN course_enrolments ce ON ce.student = s.id
JOIN courses c ON c.id = ce.course
JOIN subjects subj ON subj.id = c.subject
WHERE ce.mark >= 50
  AND subj.code LIKE 'COMP9%'
  AND EXISTS (
      SELECT 1
      FROM courses c2011
      JOIN semesters sem2011 ON sem2011.id = c2011.semester
      WHERE c2011.subject = subj.id
        AND sem2011.year = 2011
  )
GROUP BY p.id, p.unswid, p.name
HAVING COUNT(DISTINCT subj.id) >= 15;
```

```sql
CREATE OR REPLACE VIEW Q2(subject_code, mark_difference) AS
WITH pass_2008 AS (
    SELECT subj.code AS subject_code,
           AVG(ce.mark) AS avg_mark_2008,
           COUNT(*) AS pass_count_2008
    FROM course_enrolments ce
    JOIN courses c ON c.id = ce.course
    JOIN subjects subj ON subj.id = c.subject
    JOIN semesters sem ON sem.id = c.semester
    WHERE ce.mark >= 50
      AND sem.year = 2008
    GROUP BY subj.code
),
pass_2012 AS (
    SELECT subj.code AS subject_code,
           AVG(ce.mark) AS avg_mark_2012,
           COUNT(*) AS pass_count_2012
    FROM course_enrolments ce
    JOIN courses c ON c.id = ce.course
    JOIN subjects subj ON subj.id = c.subject
    JOIN semesters sem ON sem.id = c.semester
    WHERE ce.mark >= 50
      AND sem.year = 2012
    GROUP BY subj.code
)
SELECT p12.subject_code,
       ROUND((p12.avg_mark_2012 - p08.avg_mark_2008)::numeric, 2) AS mark_difference
FROM pass_2008 p08
JOIN pass_2012 p12 ON p12.subject_code = p08.subject_code
WHERE p08.pass_count_2008 >= 10
  AND p12.pass_count_2012 >= 10
  AND (p12.avg_mark_2012 - p08.avg_mark_2008) >= 10
;
```

```sql
CREATE OR REPLACE VIEW Q3(staff_id, staff_name, faculty_name) AS
WITH faculty_aff AS (
    SELECT a.staff,
           a.role,
           ou.id   AS faculty_id,
           ou.name AS faculty_name
    FROM affiliations a
    JOIN orgunits ou ON ou.id = a.orgunit
    JOIN orgunit_types ot ON ot.id = ou.utype
    WHERE ot.name = 'Faculty'
),
all_aff AS (
    SELECT staff, COUNT(*) AS total_aff_count
    FROM affiliations
    GROUP BY staff
),
faculty_only_staff AS (
    SELECT fa.staff,
           fa.faculty_id,
           fa.faculty_name,
           COUNT(*) AS faculty_aff_count,
           COUNT(DISTINCT fa.role) AS distinct_role_count
    FROM faculty_aff fa
    GROUP BY fa.staff, fa.faculty_id, fa.faculty_name
)
SELECT p.unswid, p.name, fos.faculty_name
FROM faculty_only_staff fos
JOIN all_aff aa ON aa.staff = fos.staff
JOIN people p ON p.id = fos.staff
JOIN staff s ON s.id = fos.staff
WHERE fos.faculty_aff_count = aa.total_aff_count
  AND fos.distinct_role_count >= 3
;
```

```sql
CREATE OR REPLACE VIEW Q4(program_id, unsw_id, max_wam) AS
WITH program2010 AS (
    SELECT DISTINCT pe.program
    FROM program_enrolments pe
    JOIN semesters sem ON pe.semester = sem.id
    WHERE sem.year = 2010
),
program_students AS (
    SELECT pe.student AS student_id,
      	pe.semester,
	      pe.program
    FROM program_enrolments pe
    JOIN semesters sem ON pe.semester = sem.id
    WHERE sem.year <= 2010
),
valid_course AS (
    SELECT ce.student,
           cou.semester,
		ce.course,
		cou.subject,
           ce.mark,
           subj.uoc
    FROM course_enrolments ce
    JOIN courses cou ON ce.course = cou.id
    JOIN subjects subj ON cou.subject = subj.id
    JOIN semesters sem ON cou.semester = sem.id
    WHERE sem.year <= 2010
      AND ce.mark IS NOT NULL
),
program_course_mark AS (
    SELECT ps.program,
           ps.student_id,
           vc.semester,
           vc.course,
           vc.subject,
           vc.mark,
           vc.uoc
    FROM program_students ps
    JOIN valid_course vc ON ps.student_id = vc.student
     AND ps.semester = vc.semester
),
student_program_wam AS (
    SELECT pm.program,
		pm.student_id,
		ROUND(SUM(pm.mark * pm.uoc)::numeric / NULLIF(SUM(pm.uoc), 0), 2) AS wam
    FROM program_course_mark pm
    GROUP BY pm.program, pm.student_id
    HAVING SUM(pm.uoc) > 0
),
program_max_wam AS (
    SELECT spw.program,
           MAX(spw.wam) AS max_wam
    FROM student_program_wam spw
    JOIN program2010 p10 ON p10.program = spw.program
    GROUP BY SPW.program
)
SELECT spw.program AS program_id,
       p.unswid,
       spw.wam AS max_wam
FROM student_program_wam spw
JOIN program_max_wam pmw
  ON spw.program = pmw.program
 AND spw.wam = pmw.max_wam
JOIN people p ON p.id = spw.student_id
ORDER BY spw.program, p.unswid
LIMIT 50
;
```

```sql
CREATE OR REPLACE VIEW Q5(room_id, distinct_course_count) AS
WITH candidate_rooms AS (
    SELECT cla.room,
		cou.semester,
		COUNT(DISTINCT cla.course) AS course_count,
		COUNT(DISTINCT cla.ctype) AS class_type_count
    FROM classes cla
    JOIN courses cou ON cla.course = cou.id
    GROUP BY cla.room, cou.semester
    HAVING COUNT(DISTINCT cla.course) >= 15
       AND COUNT(DISTINCT cla.ctype) >=4
),
valid_rooms AS (
    SELECT DISTINCT room
    FROM candidate_rooms
)
SELECT cla.room AS room_id,
	 COUNT(DISTINCT cla.course) AS distinct_course_count
FROM classes cla
JOIN valid_rooms vr ON vr.room = cla.room
GROUP BY cla.room
;	
```

```sql
CREATE OR REPLACE VIEW Q6(staff_id, staff_name, unique_co_teachers) AS
WITH same_course AS (
    SELECT cs1.course AS course, 
		cs1.staff AS staff1, 
		cs2.staff AS staff2
    FROM course_staff cs1
    JOIN course_staff cs2 ON cs1.course = cs2.course
    JOIN staff_roles sr1 ON sr1.id = cs1.role
    WHERE cs1.staff != cs2.staff
      AND sr1.name != 'Course Convenor'
),
valid_staff AS ( 
    SELECT sc.staff1 AS staff,
		COUNT(DISTINCT sc.staff2) AS unique_co_teachers
    FROM same_course sc
    GROUP BY sc.staff1
    HAVING COUNT(DISTINCT sc.staff2) >= 30
)
SELECT p.unswid AS staff_id,
       p.name AS staff_name,
       vs.unique_co_teachers
FROM valid_staff vs
JOIN staff s ON s.id = vs.staff
JOIN people p ON p.id = s.id
;
```

```sql
CREATE OR REPLACE FUNCTION Q7(prog_code character(4), start_year integer) RETURNS text AS $$
DECLARE
    v_program_id integer;
    v_cohort_size integer;
    v_retained_count integer;
    v_rate numeric;
BEGIN
    SELECT p.id
    INTO v_program_id
    FROM programs p
    LEFT JOIN program_enrolments pe ON pe.program = p.id
    WHERE p.code = prog_code
    GROUP by p.id
    ORDER BY MIN(pe.semester) NULLS LAST, p.id
    LIMIT 1;

    IF v_program_id IS NULL THEN
	 RETURN 'WARNING: Invalid Program or Cohort';
    END IF;


    SELECT COUNT(*)
    INTO v_cohort_size
    FROM ( 
	 SELECT pe.student
	 FROM program_enrolments pe
	 JOIN semesters sem ON sem.id = pe.semester
	WHERE pe.program = v_program_id
	GROUP BY pe.student
	HAVING MIN(sem.year) = start_year
    ) cohort;
    
    IF v_cohort_size = 0 THEN
  	RETURN 'WARNING: Invalid Program or Cohort';
    END IF;
	 

    SELECT COUNT(*)
    INTO v_retained_count
    FROM (
  	SELECT pe.student
	FROM program_enrolments pe
	JOIN semesters sem ON sem.id = pe.semester
	WHERE pe.program = v_program_id
	  AND sem.year = start_year + 2
	  AND pe.student IN (
		 	SELECT pe2.student
              		FROM program_enrolments pe2
              		JOIN semesters sem2 ON sem2.id = pe2.semester
              		WHERE pe2.program = v_program_id
              		GROUP BY pe2.student
              		HAVING MIN(sem2.year) = start_year    
	   			)

	 GROUP BY pe.student
    	 ) retained;

    
    v_rate := ROUND(v_retained_count * 100.0 / v_cohort_size, 2);

    RETURN 'Cohort size: ' || v_cohort_size || ', Retained after 2 years: ' || v_rate || '%';
END;
$$ language plpgsql;
```

```sql
CREATE OR REPLACE FUNCTION Q8(p_unswid integer, target_year integer, target_term character(2)) RETURNS text AS $$
DECLARE
    v_semester_id integer;
    v_valid_count integer;
    v_attempted_uoc integer;
    v_passed_uoc integer;
    v_term_wam numeric;

    v_prev_semester_id integer;
    v_prev_attempted_uoc integer;
    v_prev_passed_uoc integer;
    v_prev_term_wam numeric;

BEGIN
    SELECT sem.id
    INTO v_semester_id
    FROM semesters sem 
    WHERE sem.year = target_year
      AND sem.term = target_term
    ORDER BY sem.id
    LIMIT 1;
    
    
    SELECT COUNT(*)
    INTO v_valid_count
    FROM course_enrolments ce
    JOIN courses cour ON cour.id = ce.course
    JOIN people p ON p.id = ce.student
    WHERE cour.semester = v_semester_id
     AND p.unswid = p_unswid
     AND ce.mark IS NOT NULL;

    IF v_valid_count = 0 THEN
        RETURN 'No valid enrolments for term';
    END IF;

    
    SELECT COALESCE(SUM(subj.uoc), 0)
    INTO v_attempted_uoc
    FROM course_enrolments ce
    JOIN courses cour ON cour.id = ce.course
    JOIN subjects subj ON subj.id = cour.subject
    JOIN people p ON p.id = ce.student
    WHERE cour.semester = v_semester_id
     AND p.unswid = p_unswid
     AND ce.mark IS NOT NULL;

   
    SELECT COALESCE(SUM(subj.uoc), 0)
    INTO v_passed_uoc
    FROM course_enrolments ce
    JOIN courses cour ON cour.id = ce.course
    JOIN subjects subj ON subj.id = cour.subject
    JOIN people p ON p.id = ce.student
    WHERE cour.semester = v_semester_id
     AND p.unswid = p_unswid
     AND ce.mark IS NOT NULL
     AND ce.mark >= 50;

    SELECT ROUND(SUM(ce.mark * subj.uoc)::numeric / NULLIF(SUM(subj.uoc), 0),2)
    INTO v_term_wam
    FROM course_enrolments ce
    JOIN courses cour ON cour.id = ce.course
    JOIN subjects subj ON subj.id = cour.subject
    JOIN people p ON p.id = ce.student
    WHERE cour.semester = v_semester_id
     AND p.unswid = p_unswid
     AND ce.mark IS NOT NULL;
    

    IF v_term_wam >= 50 AND v_passed_uoc * 2 >= v_attempted_uoc THEN
	 RETURN 'Good Standing';
    END IF;


    SELECT MAX(cour.semester)
    INTO v_prev_semester_id
    FROM course_enrolments ce
    JOIN courses cour ON cour.id = ce.course
    JOIN people p ON p.id = ce.student
    WHERE p.unswid = p_unswid
     AND ce.mark IS NOT NULL
     AND cour.semester < v_semester_id;

    IF v_prev_semester_id IS NULL THEN
       RETURN 'Probation';
    END IF;
	

    SELECT COALESCE(SUM(subj.uoc), 0)
    INTO v_prev_attempted_uoc
    FROM course_enrolments ce
    JOIN courses cour ON cour.id = ce.course
    JOIN subjects subj ON subj.id = cour.subject
    JOIN people p ON p.id = ce.student
    WHERE cour.semester = v_prev_semester_id
      AND p.unswid = p_unswid
      AND ce.mark IS NOT NULL;

    SELECT COALESCE(SUM(subj.uoc), 0)
    INTO v_prev_passed_uoc
    FROM course_enrolments ce
    JOIN courses cour ON cour.id = ce.course
    JOIN subjects subj ON subj.id = cour.subject
    JOIN people p ON p.id = ce.student
    WHERE cour.semester = v_prev_semester_id
      AND p.unswid = p_unswid
      AND ce.mark IS NOT NULL
      AND ce.mark >= 50;

    SELECT ROUND(
           SUM(ce.mark * subj.uoc)::numeric / NULLIF(SUM(subj.uoc), 0),
           2
       )
    INTO v_prev_term_wam
    FROM course_enrolments ce
    JOIN courses cour ON cour.id = ce.course
    JOIN subjects subj ON subj.id = cour.subject
    JOIN people p ON p.id = ce.student
    WHERE cour.semester = v_prev_semester_id
      AND p.unswid = p_unswid
      AND ce.mark IS NOT NULL;

    IF v_prev_term_wam < 50 OR v_prev_passed_uoc * 2 < v_prev_attempted_uoc THEN
        RETURN 'Suspension';
    ELSE
        RETURN 'Probation';
    END IF;    
END;
$$ language plpgsql;
```

```sql
CREATE OR REPLACE FUNCTION Q9(p_unswid integer) RETURNS text AS $$
DECLARE
    v_student_id integer;
    v_stream_code text;
    v_stream_prefix text;
    v_overall_wam numeric;
    v_major_wam numeric;
    v_difference numeric;
BEGIN
    SELECT s.id
    INTO v_student_id
    FROM people p
    JOIN students s ON s.id = p.id
    WHERE p.unswid = p_unswid;

    IF v_student_id IS NULL THEN
	 RETURN 'WARNING: Invalid Student Input';
    END IF;

     
    SELECT st.code
    INTO v_stream_code
    FROM program_enrolments pe
    JOIN stream_enrolments se ON se.partof = pe.id
    JOIN streams st ON st.id = se.stream
    WHERE pe.student = v_student_id
    ORDER BY se.stream
    LIMIT 1;
    
    v_stream_prefix := LEFT(v_stream_code, 4);

    
    SELECT COALESCE(ROUND(SUM(ce.mark * subj.uoc)::numeric / NULLIF(SUM(subj.uoc), 0), 2), 0.00)
    INTO v_overall_wam
    FROM course_enrolments ce
    JOIN courses cour ON ce.course = cour.id
    JOIN subjects subj ON subj.id = cour.subject
    WHERE ce.student = v_student_id
     AND ce.mark is NOT NULL
     AND ce.grade NOT IN ('SY', 'XE', 'T', 'PE');
    

    SELECT COALESCE(ROUND(SUM(ce.mark * subj.uoc)::numeric / NULLIF(SUM(subj.uoc), 0), 2), 0.00)
    INTO v_major_wam
    FROM course_enrolments ce
    JOIN courses cour ON ce.course = cour.id
    JOIN subjects subj ON subj.id = cour.subject
    WHERE ce.student = v_student_id
     AND ce.mark is NOT NULL
     AND ce.grade NOT IN ('SY', 'XE', 'T', 'PE') 
     AND LEFT(subj.code, 4) = v_stream_prefix;
    

    v_difference := ROUND(v_major_wam - v_overall_wam, 2);

    RETURN 'Overall WAM: ' || to_char(v_overall_wam, 'FM999999990.00') ||
           ', Major WAM: ' || to_char(v_major_wam, 'FM999999990.00') ||
           ', Difference: ' ||
           CASE
               WHEN v_difference > 0
               THEN '+' || to_char(v_difference, 'FM999999990.00')
               ELSE to_char(v_difference, 'FM999999990.00')
           END;
END;
$$ language plpgsql;
```

```sql
CREATE OR REPLACE FUNCTION Q10(p_unswid integer) RETURNS SETOF text AS $$
DECLARE
    v_student_id integer;    
    rec_sem record;
    rec_course record;
    v_term_wam numeric;
    v_term_uoc_passed integer;
    v_cum_wam numeric;
    v_total_uoc integer;
BEGIN

    SELECT s.id
    INTO v_student_id
    FROM students s
    JOIN people p ON s.id = p.id
    WHERE p.unswid = p_unswid;

    IF v_student_id IS NULL THEN
	 RETURN NEXT 'WARNING: Invalid Student Input';
	 RETURN;
    END IF; 

 
    FOR rec_sem IN
	 SELECT DISTINCT sem.id, sem.year, sem.term, sem.starting
	 FROM course_enrolments ce
	 JOIN courses cour ON cour.id = ce.course
    	 JOIN semesters sem ON sem.id = cour.semester
	 WHERE ce.student = v_student_id
	 ORDER BY sem.starting, sem.id
    LOOP
	 RETURN NEXT '--- ' || rec_sem.year || ' ' || rec_sem.term || ' ---';
		FOR rec_course IN
		    SELECT subj.code,
		   	    ce.mark,
			    ce.grade
		    FROM course_enrolments ce
		    JOIN courses cour ON cour.id = ce.course
		    JOIN subjects subj ON subj.id = cour.subject
		    WHERE ce.student = v_student_id
		     AND cour.semester = rec_sem.id
		    ORDER BY subj.code
		LOOP
		    IF rec_course.mark is NULL THEN
			 RETURN NEXT rec_course.code || ' - N/A (N/A)';
		    ELSE
			 RETURN NEXT rec_course.code || ' - ' 
					|| rec_course.mark || ' (' 
					|| rec_course.grade || ')';
		    END IF;
		END LOOP;

		SELECT COALESCE(ROUND(SUM(ce.mark * subj.uoc)::numeric / NULLIF(SUM(subj.uoc), 0), 2), 0.00)		
		INTO v_term_wam
     		FROM course_enrolments ce
    		JOIN courses cour ON cour.id = ce.course
		JOIN subjects subj ON subj.id = cour.subject
		WHERE ce.student = v_student_id
		AND cour.semester = rec_sem.id
		AND ce.mark IS NOT NULL
		AND ce.grade NOT IN ('SY', 'XE', 'T', 'PE');
		
		SELECT COALESCE(SUM(subj.uoc), 0)
		INTO v_term_uoc_passed
		FROM course_enrolments ce
		JOIN courses cour ON cour.id = ce.course
		JOIN subjects subj ON subj.id = cour.subject
		WHERE ce.student = v_student_id
		AND cour.semester = rec_sem.id
		AND ce.mark >= 50;
		
		RETURN NEXT 'Term WAM: ' || to_char(v_term_wam, 'FM999999990.00') ||
                ' | Term UOC Passed: ' || v_term_uoc_passed;
    END LOOP;
    
    SELECT COALESCE(
           ROUND(
               SUM(ce.mark * subj.uoc)::numeric / NULLIF(SUM(subj.uoc), 0),
               2
           ),
           0.00
       )
    INTO v_cum_wam
    FROM course_enrolments ce
    JOIN courses cour ON cour.id = ce.course
    JOIN subjects subj ON subj.id = cour.subject
    WHERE ce.student = v_student_id
     AND ce.mark IS NOT NULL
     AND ce.grade NOT IN ('SY', 'XE', 'T', 'PE');
 
    SELECT COALESCE(SUM(subj.uoc), 0)
    INTO v_total_uoc
    FROM course_enrolments ce
    JOIN courses cour ON cour.id = ce.course
    JOIN subjects subj ON subj.id = cour.subject
    WHERE ce.student = v_student_id
     AND ce.mark >= 50;
    
    RETURN NEXT '=== CUMULATIVE WAM: ' || to_char(v_cum_wam, 'FM999999990.00') ||
            ' | TOTAL UOC: ' || v_total_uoc || ' ===';

    RETURN;
END;

$$ language plpgsql;
```
